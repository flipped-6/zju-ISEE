/**
  * @brief  The application entry point.
  * @retval int
  */
int main(void)
{

  /* USER CODE BEGIN 1 */
	uint16_t adc_value;//ADC采集值
  uint8_t oled_buf[16];
  /* USER CODE END 1 */

  /* MCU Configuration--------------------------------------------------------*/

  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
  HAL_Init();

  /* USER CODE BEGIN Init */

  /* USER CODE END Init */

  /* Configure the system clock */
  SystemClock_Config();

  /* USER CODE BEGIN SysInit */

  /* USER CODE END SysInit */

  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  MX_TIM2_Init();
  MX_USART1_UART_Init();
  MX_ADC1_Init();
  MX_I2C2_Init();
  /* USER CODE BEGIN 2 */
	HAL_UART_Receive_IT(&huart1,&RxByte,1);
	HAL_TIM_PWM_Start(&htim2,TIM_CHANNEL_2);
	HAL_TIM_PWM_Start(&htim2,TIM_CHANNEL_3);
	HAL_TIM_PWM_Start(&htim2,TIM_CHANNEL_4);//PWM值分别映射到对应通道
	
  Oled_Init();
  OLED_Clear(0x00);//清屏准备
 


	
  /* USER CODE END 2 */

  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
  while (1)
  {
    /* USER CODE END WHILE */

    /* USER CODE BEGIN 3 */
		switch (KeyV)
		{
			case KEY0_PRES:
			{
				//srand((unsigned int)time(NULL)); 
				 random_number1 = rand()%999;//random bright
				 random_number2 = rand()%999;
				 random_number3 = rand()%999;
						break;
			}
			default:
				HAL_Delay(10);
		}
		
		flag=1;
		KeyV = KEY_Scan(0);
		if(!KEY2)
			{
				if(num>2){
					flag=0;
				 random_number1 = rand()%999;//赋予随机亮度
				 random_number2 = rand()%999;
				 random_number3 = rand()%999;
				}	
					
      }
		num++;if(num>4)num=0;//闪烁周期控制

    if(WK_UP){//按键控制亮度增加
      scl++;
    }
    if(!KEY1){//按键控制亮度减少
      scl--;
    }
		
    //开始ADC采集
		HAL_ADC_Start(&hadc1);
		HAL_ADC_PollForConversion(&hadc1,10);
		adc_value= HAL_ADC_GetValue(&hadc1);
		printf("Voltage=%.3fV\r\n",(float)adc_value*3.3/4096.0);
    scl2=100*adc_value/4096;
    sprintf(oled_buf,"Voltage=%.3fV",(float)adc_value*3.3/4096.0);
    Oled_Display_String(0,0,oled_buf);
		
    if(scl>scl2){
      scl_final=scl2;
    }else{
      scl_final=scl;
    }//取串口设置和电位器设置的最小值

    //SCL限幅
		if(scl_final>=100)scl_final=100;
		if(scl_final<=0)scl_final=0;
    //执行PWM输出
    if(flag){
      __HAL_TIM_SET_COMPARE(&htim2,TIM_CHANNEL_2,(uint16_t)random_number1*scl_final/100);
      __HAL_TIM_SET_COMPARE(&htim2,TIM_CHANNEL_3,(uint16_t)random_number2*scl_final/100);
      __HAL_TIM_SET_COMPARE(&htim2,TIM_CHANNEL_4,(uint16_t)random_number3*scl_final/100);
    }else{
      __HAL_TIM_SET_COMPARE(&htim2,TIM_CHANNEL_2,0);
      __HAL_TIM_SET_COMPARE(&htim2,TIM_CHANNEL_3,0);
      __HAL_TIM_SET_COMPARE(&htim2,TIM_CHANNEL_4,0);
    } 
		
		
		if(uartRx_flag)//串口指令通信
		{
			uartRx_flag = 0;
			RxCount =0;
			if(Compare_string("AT+ON"))
			{
				//printf("Voltage=%4d\r\n",RxBuf[1]);
				if(scl)scl = 0;
				else scl = 100;
			}
	  		
		  else if(Compare_string("AT+BRIGHT="))	//
			{		
				//printf("Voltage=%4d\r\n",RxBuf[10]);
				scl = ((int)RxBuf[10]-48)*10+(int)RxBuf[11]-48;
			}
			else if(Compare_string("AT+COLOR="))
			{
				switch (RxBuf[9])
				{
					case 'R':
					{
						//printf("Voltage=%4d\r\n",RxBuf[1]);
						random_number1 = 999;
						random_number2 = 0;
						random_number3 = 0;
						break;
					}
					case 'G':
					{
						random_number1 = 0;
						random_number2 = 999;
						random_number3 = 0;
						break;
					}
					case 'B':
					{
						random_number1 = 0;
						random_number2 = 0;
						random_number3 = 999;
						break;
					}
					default:
					{
						random_number1 = 999;
						random_number2 = 999;
						random_number3 = 999;
					}
				}
			}
		//Tone(50);
		for (int j = 0; j < 32; ++j)								//
			{
			receiveData[j] =0;
			}
		}
		
		
		sprintf(oled_buf,"Red:  %6.2f%%",(float)random_number1*(float)scl_final/1000.0);
    Oled_Display_String(2,0,oled_buf);
		sprintf(oled_buf,"Green:%6.2f%%",(float)random_number2*(float)scl_final/1000.0);
    Oled_Display_String(4,0,oled_buf);
		sprintf(oled_buf,"Blue: %6.2f%%",(float)random_number3*(float)scl_final/1000.0);
    Oled_Display_String(6,0,oled_buf);//屏幕显示RGB相对值
	}
  /* USER CODE END 3 */
}

  RCC_ClkInitTypeDef RCC_ClkInitStruct = {0};
  RCC_PeriphCLKInitTypeDef PeriphClkInit = {0};

  /** Initializes the RCC Oscillators according to the specified parameters
  * in the RCC_OscInitTypeDef structure.
  */
  RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_HSE;
  RCC_OscInitStruct.HSEState = RCC_HSE_ON;
  RCC_OscInitStruct.HSEPredivValue = RCC_HSE_PREDIV_DIV1;
  RCC_OscInitStruct.HSIState = RCC_HSI_ON;
  RCC_OscInitStruct.PLL.PLLState = RCC_PLL_ON;
  RCC_OscInitStruct.PLL.PLLSource = RCC_PLLSOURCE_HSE;
  RCC_OscInitStruct.PLL.PLLMUL = RCC_PLL_MUL2;
  if (HAL_RCC_OscConfig(&RCC_OscInitStruct) != HAL_OK)
  {
    Error_Handler();
  }

  /** Initializes the CPU, AHB and APB buses clocks
  */
  RCC_ClkInitStruct.ClockType = RCC_CLOCKTYPE_HCLK|RCC_CLOCKTYPE_SYSCLK
                              |RCC_CLOCKTYPE_PCLK1|RCC_CLOCKTYPE_PCLK2;
  RCC_ClkInitStruct.SYSCLKSource = RCC_SYSCLKSOURCE_PLLCLK;
  RCC_ClkInitStruct.AHBCLKDivider = RCC_SYSCLK_DIV2;
  RCC_ClkInitStruct.APB1CLKDivider = RCC_HCLK_DIV2;
  RCC_ClkInitStruct.APB2CLKDivider = RCC_HCLK_DIV1;
